uses GraphABC;
begin
  SetWindowSize(320, 340);

  var n: integer; write('введите кол-во отверстий до 8: '); read(n);

  SetBrushColor(clGray); Circle(160, 170, 150);
  SetBrushColor(clWhite); Circle(160, 170, 50);

  for var i:=0 to n-1 do begin
    var deg := 45*i;
    var x := 160 + 100*cos(deg*pi/180);
    var y := 170 - 100*sin(deg*pi/180);
    SetBrushColor(clWhite); Circle(Round(x),Round(y), 30);
  end;
end.
