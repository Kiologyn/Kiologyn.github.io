uses GraphABC;
var n, x, y, i: integer;
var deg: real;
begin
  SetWindowSize(320, 340);
    
  write('введите кол-во отверстий до 8: ');
  read(n);
    
  SetBrushColor(clGray);
  Circle(160, 170, 150);
  SetBrushColor(clWhite);
  Circle(160, 170, 50);
    
  for i:=0 to n-1 do begin
    deg := -(360/n*i)+90;
    x := Ceil(160 + 100*cos(deg*pi/180));
    y := Ceil(170 - 100*sin(deg*pi/180));
    Circle(x,y, 30);
  end;
end.
